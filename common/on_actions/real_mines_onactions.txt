on_actions = {
	on_startup = {
		effect = {
			every_country = {
                reset_mines = yes
			}
		}
	}
    
    on_daily = {
        effect = {
            if = { 
                limit = { check_variable = { started = 1 } }
                add_to_variable = { timer = 1 }
                update_timer_text = yes
            }
        }
    }
}
