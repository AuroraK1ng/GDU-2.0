# Finland Refuses Demands

	id = soviet.13
	title = soviet.13.t
	desc = soviet.13.d
	picture = GFX_report_event_soviet_tanks_snow

	fire_only_once = yes
	trigger = {
		tag = SOV
		has_global_flag = soviet_demand
		NOT = { has_global_flag = finland_refuse }

			}
		}
	}
